        self.mod1 = mod1
        self.x2 = x2
        self.mod2 = mod2
        self.x3 = x3
        self.mod3 = mod3

        return self.mod1 * self.mod2 * self.mod3

def kitrsa(N1, N2, N3, C1, C2, C3):
    Y = N1 * N2 * N3
    y1 = N2 * N3
    y2 = N1 * N3
    y3 = N1 * N2
    n1 = pow(y1, -1, N1)
    n2 = pow(y2, -1, N2)
    n3 = pow(y3, -1, N3)

    M = round(pow(C, 1/3))
